 62def load_lazy_content(driver, scroll_step=config["scraping"]["lazy_scroll_step"],
 63                      wait_time=config["scraping"]["wait_time"]):
 64    """Scroll epika.lrt.lt page to load lazy content"""
 65
 66    # Get the current scroll position
 67    current_position = driver.execute_script("return window.pageYOffset || document.documentElement.scrollTop")
 68    logger.info("Current position: %s", current_position)
 69
 70    # Find max height of the page
 71    max_height = driver.execute_script("return document.body.scrollHeight")
 72    logger.info("Max height: %s", max_height)
 73
 74    # Step by step scroll page while content downloads
 75    logger.info("Scrolling...")
 76    for scroll_position in range(current_position, max_height, scroll_step):
 77        print(f"Scroll position: {scroll_position}", end='\r')
 78        driver.execute_script(f"window.scrollTo(0, {scroll_position});")
 79        time.sleep(wait_time)
 80    print("\n")
 81
 82    # Final scroll to the bottom to ensure all content is loaded
 83    driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
 84    time.sleep(wait_time)

100    """Scrape epika.lrt.lt page for media information based on the list of search strings."""
101
102    logging.info("Starting shallow scraping...")
103    # Open web page for the first time and accept the cookies
104    driver.get("https://epika.lrt.lt/search")
105    accept_cookies(driver)
106    logging.info("Cookies accepted")
107
108    # Initialize the list of movies for return as function result
109    list_of_movies: list[tuple[str, str, str]] = []
110
111    # In demo mode perform less movie searches and scrape less pages
112    if config["demo"]["is_demo"]:

115        list_search_strings_epika = LargeStrings.list_search_strings_epika
116
117    # Loop through all search strings
118    for ind, search_string in enumerate(list_search_strings_epika, start=1):
119        logging.info("Shallow scraping - page %s of %s", ind, len(LargeStrings.list_search_strings_epika))
120        counter_str_used = 0  # To count additions in relation to search string
121        try:
122            # Open the webpage
123            driver.get(f"https://epika.lrt.lt/search?q={search_string}")
124
125            time.sleep(2)  # Allow to load whole body of the page
126
127            # Easy scroll the page to the bottom to download its content
128            load_lazy_content(driver)
129
130            # Find all media blocks
131            title_blocks: list[WebElement] = driver.find_elements(By.CSS_SELECTOR, ".tile--vod.tile")
132            logging.info("Found %s movie title. Extracting...", len(title_blocks))
133
134            for i, block in enumerate(title_blocks):
135                try:
136                    # Extract title, link to page, and link to image
137                    movie_title = block.find_element(By.CSS_SELECTOR, ".headline-4.tile__title").text
138                    link_to_page = block.find_element(By.CLASS_NAME, "tile__link").get_attribute("href")
139                    link_to_image = block.find_element(By.CLASS_NAME, "cover").get_attribute("src")
140                    print("." * i, end='\r')
141
142                    # No way further with this movie, if one element is missing
143                    assert movie_title is not None, "Movie title is None"
144                    assert link_to_page is not None, "Link to page is None"
145                    assert link_to_image is not None, "Link to image is None"
146                    # Check if the movie title is already in the list
147                    if not any(link_to_page == existing_url for _, existing_url, _ in list_of_movies):
148                        # Add the tuple to the list
149                        counter_str_used += 1
150                        list_of_movies.append((movie_title, link_to_page, link_to_image))
151
152                except NoSuchElementException as err:
153                    logging.warning("Element not found: %s", err)
154
155        except Exception:
156            logging.exception(f"An error occurred while processing '{search_string}'.", search_string)
157
158        print(f'\nString: "{search_string}" | Returns: {len(title_blocks)} | Used: {counter_str_used}\n\n')
159        time.sleep(2)  # Make pause between scraping next page
160
161    logging.info("Shallow scraping finished.")
162    return list_of_movies

165def deep_scrape_epika(driver: webdriver.Chrome, list_of_movies: list[tuple[str, str, str]]) -> list[
166    tuple[str, bytes, str, int, int, str, str]]:
167    """Scrape epika.lrt.lt particular movie page for additional information of the movie."""
168
169    logging.info("Starting deep scraping...")
170    # Open web page for the first time and accept the cookies
171    driver.get("https://epika.lrt.lt/search")
172    time.sleep(2)
173    accept_cookies(driver)
174    logging.info("Cookies accepted")
175
176    # Initialize the list of movie data for return as function result
177    # Tuple structure: <title, image, description, release year, duration, genre, page url>
178    list_of_movie_data: list[tuple[str, Optional[bytes], str, int, int, str, str]] = []
179
180    for ind, movie in enumerate(list_of_movies, start=1):
181        print(f"Scraping {ind} of {len(list_of_movies)}", end='\r')
182        try:
183            driver.get(f"{movie[1]}")
184            time.sleep(1)  # Allow the page to load
185
186            # Initialize variables
187            release_year = None
188            genre = []
189            total_minutes = None
190
191            try:
192                metadata_container = driver.find_element(By.CSS_SELECTOR, 'div.metadata__product-meta')
193                metadata_elements = metadata_container.find_elements(By.CSS_SELECTOR,
194                                                                     'span.metadata__product-meta-element')
195
196                for element in metadata_elements:
197                    text = element.text.strip()
198
199                    if text.isdigit() and len(text) == 4:
200                        release_year = int(text)
201                    elif re.match(r'(?:(\d+)h\s*)?(\d+)m', text):
202                        match = re.match(r'(?:(\d+)h\s*)?(\d+)m', text)
203                        hours, minutes = map(lambda x: int(x) if x else 0, match.groups())
204                        total_minutes = hours * 60 + minutes
205                    else:
206                        genre.append(text)
207
208                genre = ', '.join(genre)
209
210            except Exception as e:
211                logging.warning("Error extracting metadata: %s", e)
212
213            try:
214                description = driver.find_element(By.CSS_SELECTOR, 'div.metadata-content__description').text.strip()
215            except NoSuchElementException as err:
216                logging.warning("Description not found: %s", err)
217                description = ""
218
219            image = read_image_from_url(movie[2])
220
221            # Append the movie data to the list
222            list_of_movie_data.append((movie[0], image, description, release_year, total_minutes, genre, movie[1]))
223
224            # Assertions
225            assert release_year is not None, "Release year is None"
226            assert genre != "", "Genre is None"
227            assert total_minutes is not None, "Duration is None"
228            assert description != "", "Description is None"
229            assert image is not None, "Image is None"
230
231        except Exception as e:
232            logging.info("Element not found '%s': %s", movie[0], e)
233
234        time.sleep(1)  # Pause between scraping pages
235
236    logging.info("Deep scraping finished.")
237    return list_of_movie_data

287def convert_duration_to_minutes(duration_str: str) -> int | None | Any:
288    """Converts duration strings from lrt.lt/tema/filmai page to minutes"""
289
290    parts = duration_str.split(":")
291    try:
292        if len(parts) == 3:
293            # Format is HH:MM:SS
294            hours, minutes, _ = map(int, parts)
295            return hours * 60 + minutes
296        elif len(parts) == 2:
297            # Format is MM:SS
298            minutes, _ = map(int, parts)
299            return minutes
300    except ValueError:
301        return None
302    return None

333            except (NoSuchElementException, ElementNotInteractableException):

336
337        # Find all media blocks
338        news_blocks = driver.find_elements(By.CLASS_NAME, "news")
339        logging.info("Blocks loaded: %s", len(news_blocks))
340
341        # Initialize a list to store the tuples for return as function result
342        media_info = []
343
344        # Loop through each news block
345        for ind, block in enumerate(news_blocks):
346            print(f"Processing block: {ind + 1}", end='\r')
347
348            # Check if the specific icon element exists, skip if it does - not movie
349            if block.find_elements(By.CSS_SELECTOR, "svg.svg-icon.badge-light") or block.find_elements(By.CSS_SELECTOR,
350                                                                                                       "i.icon.icon"
351                                                                                                       "-photo"):
352                continue
353
354            # Extract the title and link
355            title_element = block.find_element(By.CSS_SELECTOR, "h3.news__title a")
356            title = title_element.text
357            link = title_element.get_attribute("href")
358
359            # Extract image link
360            image_link = block.find_element(By.CSS_SELECTOR, ".media-block__image").get_attribute(
361                "src")
362
363            # Extract the duration
364            duration = block.find_element(By.CLASS_NAME, "media-block__duration").text if block.find_elements(
365                By.CLASS_NAME, "media-block__duration") else "None"
366
367            # Extract the count of views
368            views = block.find_element(By.CSS_SELECTOR,
369                                       ".badge-list.media-block__badge-list .badge.badge-light > span:last-child").text if block.find_elements(
370                By.CSS_SELECTOR, ".badge-list.media-block__badge-list .badge.badge-light > span:last-child") else "None"
371
372            # Check if both duration and views do not exist, skip - not a movie
373            if duration == 'None' and views == 'None':
374                continue
375
376            # Add the tuple to the list
377            media_info.append((title, link, image_link, duration, views))
378
379        print("\n")
380        logging.info("Shallow scraping finished.")
381        return media_info
382
383    except Exception:
384        logging.exception("An error occurred during scraping")
385        return []
